# Dropdown Menu

Animated overlay menu triggered by any element. Supports icons, disabled items, danger actions, and dividers with configurable color themes and alignment.

```tsx
import { DropdownMenu } from "@mattloyed/eva-ui";
```

## Basic Usage

<ComponentPreview
  title="DEFAULT"
  code={`import { DropdownMenu, Button } from "@mattloyed/eva-ui";

<DropdownMenu
  trigger={<Button size="sm">ACTIONS</Button>}
  items={[
    { label: "View Details", onClick: () => {} },
    { label: "Edit Record", onClick: () => {} },
    { label: "Duplicate", onClick: () => {} },
    { label: "Export Data", onClick: () => {} },
  ]}
/>`}
>
  <DropdownMenu
    trigger={<Button size="sm">ACTIONS</Button>}
    items={[
      { label: "View Details" },
      { label: "Edit Record" },
      { label: "Duplicate" },
      { label: "Export Data" },
    ]}
  />
</ComponentPreview>
