# PhaseStatusStack

A vertical stack of phase status rows with color-coded backgrounds. Each row displays a label, status color, and optional value. Used for system readiness panels, checklist displays, and multi-phase status overviews.

```tsx
import { PhaseStatusStack } from "@mdrbx/nerv-ui";
```

## Basic Usage

<ComponentPreview
  title="DEFAULT"
  code={`import { PhaseStatusStack } from "@mdrbx/nerv-ui";

<PhaseStatusStack
  title="SYSTEM CHECK"
  phases={[
    { label: "POWER", status: "ok" },
    { label: "COMMS", status: "ok" },
    { label: "WEAPONS", status: "warning" },
    { label: "SHIELDS", status: "danger" },
  ]}
/>`}
>
  <PhaseStatusStack
    title="SYSTEM CHECK"
    phases={[
      { label: "POWER", status: "ok" },
      { label: "COMMS", status: "ok" },
      { label: "WEAPONS", status: "warning" },
      { label: "SHIELDS", status: "danger" },
    ]}
  />
</ComponentPreview>

## Props

### PhaseStatusStackProps

<PropsTable
  props={[
    {
      name: "phases",
      type: "PhaseItem[]",
      description: "Array of phase items to display",
      required: true,
    },
    {
      name: "title",
      type: "string",
      description: "Title displayed above the stack",
    },
    {
      name: "color",
      type: '"orange" | "green" | "cyan"',
      default: '"green"',
      description: "Color theme for the title",
    },
    {
      name: "className",
      type: "string",
      default: '""',
      description: "Additional CSS classes",
    },
  ]}
/>

### PhaseItem

<PropsTable
  props={[
    {
      name: "label",
      type: "string",
      description: "Text label for the phase row",
      required: true,
    },
    {
      name: "status",
      type: '"ok" | "warning" | "danger" | "inactive"',
      description: "Status determining the row background color",
      required: true,
    },
    {
      name: "value",
      type: "string",
      description: "Optional value displayed on the right side of the row",
    },
  ]}
/>

<Callout type="info" title="STATUS COLORS">
  Each status maps to an EVA palette color: **ok** renders green, **warning**
  renders orange, **danger** renders red, and **inactive** renders a dimmed
  gray.
</Callout>

<Callout type="info" title="HOVER LABELS">
  Hovering or focusing the center status bar applies a light motion cue,
  switches to a pointer cursor, and surfaces the phase label, plus its value
  when one is provided.
</Callout>
