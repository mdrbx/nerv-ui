# Select Menu

Styled dropdown with angle brackets `< >` on focus. Custom arrow indicator. Extends standard HTML select attributes.

```tsx
import { SelectMenu } from "@mattloyed/eva-ui";
```

## Basic Usage

<ComponentPreview
  title="DEFAULT"
  code={`<SelectMenu
  label="PRIORITY LEVEL"
  options={[
    { value: "alpha", label: "ALPHA" },
    { value: "beta", label: "BETA" },
    { value: "gamma", label: "GAMMA" },
    { value: "omega", label: "OMEGA" },
  ]}
  color="orange"
/>`}
>
  <SelectMenu
    label="PRIORITY LEVEL"
    options={[
      { value: "alpha", label: "ALPHA" },
      { value: "beta", label: "BETA" },
      { value: "gamma", label: "GAMMA" },
      { value: "omega", label: "OMEGA" },
    ]}
    color="orange"
  />
</ComponentPreview>
