# Pagination

LCD-style page navigation with `<<` / `>>` arrow buttons and zero-padded page counter. Buttons disable automatically at boundaries.

```tsx
import { Pagination } from "@mdrbx/nerv-ui";
```

## Basic Usage

<Callout type="info" title="INTERACTIVE">
  Click the arrow buttons to navigate between pages.
</Callout>

<ComponentPreview
  title="DEFAULT"
  code={`import { useState } from "react";
import { Pagination } from "@mdrbx/nerv-ui";

const [page, setPage] = useState(1);

<Pagination
  total={120}
  pageSize={10}
  currentPage={page}
  onPageChange={setPage}
/>`}
>
  <PaginationBasicDemo />
</ComponentPreview>
