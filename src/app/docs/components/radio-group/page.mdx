# Radio Group

Terminal-style radio buttons with `(*)` / `( )` indicators. Supports vertical and horizontal layout, group labels, and disabled options.

```tsx
import { RadioGroup } from "@mattloyed/eva-ui";
```

## Basic Usage

<Callout type="info" title="INTERACTIVE">
  Click an option to change the selection.
</Callout>

<ComponentPreview
  title="VERTICAL (DEFAULT)"
  code={`import { useState } from "react";
import { RadioGroup } from "@mattloyed/eva-ui";

const [value, setValue] = useState("alpha");

<RadioGroup
  label="PRIORITY LEVEL"
  options={[
    { value: "alpha", label: "ALPHA" },
    { value: "beta", label: "BETA" },
    { value: "omega", label: "OMEGA" },
  ]}
  value={value}
  onChange={setValue}
/>`}
>
  <RadioGroupBasicDemo />
</ComponentPreview>
